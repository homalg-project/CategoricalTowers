gap> compiled_category_name := "FinBouquetsPrecompiled";;
gap> package_name := "FunctorCategories";;

>                      "InterpretMorphismFromDistinguishedObjectToHomomorphismStructureAsMorphism",
>                      "MorphismsOfExternalHom",
>                      ]
> );;
